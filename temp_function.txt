                  padding: 4px; border-radius: 3px;
                " title="View Details">ğŸ”</button>
              </div>
            </div>`;
  }
  
  // MCQ with all correct answers

                " title="View Details">ğŸ”</button>
              </div>
            </div>`;
  }
